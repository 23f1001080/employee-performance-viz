import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

np.random.seed(42)

departments = ["Operations","Sales","IT","Finance","Marketing","Logistics","Support","HR"]
counts = np.random.randint(5, 25, size=len(departments))
for d, c in zip(departments, counts):
    print(f"{d}: {c}")

ops = dict(zip(departments, counts)).get("Operations", 0)
print("Operations count =", ops)

plt.bar(departments, counts)
plt.xticks(rotation=45)
plt.title("Department Distribution")
plt.tight_layout()
plt.savefig("departments_hist.png")
